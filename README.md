## Install

```bash
npm install --save redux-helper-np
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'redux-helper-np'
import 'redux-helper-np/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
